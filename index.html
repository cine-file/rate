function doPost(e) {
  var ss = SpreadsheetApp.openById("YOUR_SHEET_ID");
  var d  = JSON.parse(e.postData.contents);
  var user = d.user || "Unknown";

  // ── Personal tab ──
  var tab = ss.getSheetByName(user) || ss.insertSheet(user);
  if (tab.getLastRow() === 0) {
    tab.appendRow(["Date","Title","Year","Director","RT Audience",
      "IMDb","Score /10","Raw /100","Grade",
      "Plot","Plot Grade","Plot Notes",
      "Entertainment","Ent Grade","Ent Notes",
      "Acting","Acting Grade","Acting Notes",
      "Visuals","Visuals Grade","Visuals Notes",
      "Pacing","Pacing Grade","Pacing Notes",
      "Emotional","Emotional Grade","Emotional Notes",
      "Overall Notes"]);
  }
  tab.appendRow([d.date,d.title,d.year,d.director,d.rt,d.imdb,
    d.score10,d.score100,d.grade,
    d.plot,d.plotGrade,d.plotNotes,
    d.entertainment,d.entGrade,d.entNotes,
    d.acting,d.actingGrade,d.actingNotes,
    d.visuals,d.visualsGrade,d.visualsNotes,
    d.pacing,d.pacingGrade,d.pacingNotes,
    d.emotional,d.emotionalGrade,d.emotionalNotes,
    d.notes]);

  // ── Summary tab ──
  var sum = ss.getSheetByName("Summary") || ss.insertSheet("Summary");
  if (sum.getLastRow() === 0)
    sum.appendRow(["Title","Year","Director","RT Audience","IMDb"]);

  // Find or create row for this title
  var data = sum.getDataRange().getValues();
  var titleCol = 0, rowIdx = -1;
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === d.title) { rowIdx = i + 1; break; }
  }
  if (rowIdx === -1) {
    sum.appendRow([d.title,d.year,d.director,d.rt,d.imdb]);
    rowIdx = sum.getLastRow();
    data = sum.getDataRange().getValues();
  }

  // Find or create column for this user
  var headers = data[0];
  var userCol = headers.indexOf(user);
  if (userCol === -1) {
    userCol = headers.length;
    sum.getRange(1, userCol + 1).setValue(user);
  }
  sum.getRange(rowIdx, userCol + 1).setValue(d.score10);

  return ContentService.createTextOutput(
    JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  var ss   = SpreadsheetApp.openById("YOUR_SHEET_ID");
  var user = e.parameter.user;
  var tab  = ss.getSheetByName(user);
  if (!tab) return ContentService.createTextOutput(
    JSON.stringify([])).setMimeType(ContentService.MimeType.JSON);
  var rows = tab.getDataRange().getValues();
  var keys = rows[0];
  var out  = [];
  for (var i = 1; i < rows.length; i++) {
    var obj = {};
    keys.forEach(function(k,j){ obj[k] = rows[i][j]; });
    out.push(obj);
  }
  return ContentService.createTextOutput(
    JSON.stringify(out)).setMimeType(ContentService.MimeType.JSON);
}
Replace YOUR_SHEET_ID with your Sheet's ID from the URL.
