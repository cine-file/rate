var SHEET_ID = "YOUR_SHEET_ID"; // ← replace this

function doPost(e) {
  var ss   = SpreadsheetApp.openById(SHEET_ID);
  var d    = JSON.parse(e.postData.contents);
  var action = d.action || "saveRating";

  // ── Save Users ──
  if (action === "saveUsers") {
    var uTab = ss.getSheetByName("Users") || ss.insertSheet("Users");
    uTab.clearContents();
    uTab.appendRow(["name","pin"]);
    // Format PIN column as text so leading zeros are preserved
    var pinCol = uTab.getRange(1, 2, 1, 1);
    pinCol.setNumberFormat("@");
    (d.users||[]).forEach(function(u){
      var row = uTab.getLastRow() + 1;
      uTab.appendRow([u.name, "'"+String(u.pin)]);
    });
    return ContentService.createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // ── Save Rating ──
  var user = d.user || "Unknown";
  var tab  = ss.getSheetByName(user) || ss.insertSheet(user);
  if (tab.getLastRow() === 0) {
    tab.appendRow(["Date","Title","Year","Director","RT Audience",
      "IMDb","Score /10","Raw /100","Grade",
      "Plot","Plot Grade","Plot Notes",
      "Entertainment","Ent Grade","Ent Notes",
      "Acting","Acting Grade","Acting Notes",
      "Visuals","Visuals Grade","Visuals Notes",
      "Pacing","Pacing Grade","Pacing Notes",
      "Emotional","Emotional Grade","Emotional Notes","Overall Notes"]);
  }
  // Check if this movie was already rated — overwrite if so
  var newRow = [d.date,d.title,d.year,d.director,d.rt,d.imdb,
    d.score10,d.score100,d.grade,
    d.plot,d.plotGrade,d.plotNotes,
    d.entertainment,d.entGrade,d.entNotes,
    d.acting,d.actingGrade,d.actingNotes,
    d.visuals,d.visualsGrade,d.visualsNotes,
    d.pacing,d.pacingGrade,d.pacingNotes,
    d.emotional,d.emotionalGrade,d.emotionalNotes,d.notes];
  var tabData = tab.getDataRange().getValues();
  var existingRow = -1;
  for (var i = 1; i < tabData.length; i++) {
    if (String(tabData[i][1]).toLowerCase() === String(d.title).toLowerCase()) {
      existingRow = i + 1; break;
    }
  }
  if (existingRow > -1) {
    tab.getRange(existingRow, 1, 1, newRow.length).setValues([newRow]);
  } else {
    tab.appendRow(newRow);
  }

  // ── Summary tab ──
  var sum  = ss.getSheetByName("Summary") || ss.insertSheet("Summary");
  if (sum.getLastRow() === 0)
    sum.appendRow(["Title","Year","Director","RT Audience","IMDb"]);
  var data = sum.getDataRange().getValues();
  var rowIdx = -1;
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] === d.title) { rowIdx = i + 1; break; }
  }
  if (rowIdx === -1) {
    sum.appendRow([d.title,d.year,d.director,d.rt,d.imdb]);
    rowIdx = sum.getLastRow();
    data = sum.getDataRange().getValues();
  }
  var headers = data[0];
  var userCol = headers.indexOf(user);
  if (userCol === -1) {
    userCol = headers.length;
    sum.getRange(1, userCol + 1).setValue(user);
  }
  sum.getRange(rowIdx, userCol + 1).setValue(d.score10);

  return ContentService.createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  var ss     = SpreadsheetApp.openById(SHEET_ID);
  var action = e.parameter.action || "getRatings";

  // ── Get Users ──
  if (action === "getUsers") {
    var uTab = ss.getSheetByName("Users");
    if (!uTab) return out([]);
    var rows = uTab.getDataRange().getValues();
    var users = [];
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][0]) users.push({name: String(rows[i][0]), pin: String(rows[i][1]).replace(/^'/,'').padStart(4,'0')});
    }
    return ContentService.createTextOutput(JSON.stringify({users:users}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // ── Get Summary (group stats) ──
  if (action === "getSummary") {
    var sum = ss.getSheetByName("Summary");
    if (!sum) return ContentService.createTextOutput(JSON.stringify({rows:[]}))
      .setMimeType(ContentService.MimeType.JSON);
    var data    = sum.getDataRange().getValues();
    var headers = data[0];
    var rows    = [];
    for (var i = 1; i < data.length; i++) {
      var row = {Title:data[i][0],Year:data[i][1],scores:[],userScores:{}};
      for (var j = 5; j < headers.length; j++) {
        if (headers[j] && data[i][j] !== "") {
          var s = parseFloat(data[i][j]);
          if (!isNaN(s)) {
            row.scores.push(s);
            row.userScores[headers[j]] = s;
          }
        }
      }
      rows.push(row);
    }
    return ContentService.createTextOutput(JSON.stringify({rows:rows}))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // ── Get Personal Ratings ──
  var user = e.parameter.user;
  var tab  = ss.getSheetByName(user);
  if (!tab) return ContentService.createTextOutput(JSON.stringify([]))
    .setMimeType(ContentService.MimeType.JSON);
  var rows = tab.getDataRange().getValues();
  var keys = rows[0];
  var out  = [];
  for (var i = 1; i < rows.length; i++) {
    var obj = {};
    keys.forEach(function(k,j){ obj[k] = rows[i][j]; });
    out.push(obj);
  }
  return ContentService.createTextOutput(JSON.stringify(out))
    .setMimeType(ContentService.MimeType.JSON);
}
